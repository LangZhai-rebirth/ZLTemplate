ZLTemplate by LangZhai(智能小菜菜)

这是啥

一款基于jQuery的轻量级模板引擎。

能干啥

填充模板，支持无限循环嵌套及灵活的传参方式。

怎么用

引入JS：

<!DOCTYPE HTML>
<html>
    <head>
        <title>ZLTemplate by LangZhai(智能小菜菜)</title>
    </head>
    <body>
        <!--为了使页面加载更快，最好将JS放置在<body>的末尾-->
        <script src="jquery.min.js"></script>
        <script src="jquery.zltemplate.min.js"></script>
    </body>
</html>

定义模板：

<!--填充目标-->
<div id="target" style="background:#272822;color:#f8f8f2;padding:1em;margin:.5em 0;overflow:auto;border-radius:.3em;"></div>

<!--主模板-->
<script type="text/html" id="template">
    <h1>#{data.title}</h1>
    <p>#{data.desc}</p>
    #{data.item === undefined ? '' : '<ul>' + nested.item + '</ul>'}
    #{data.attr === undefined ? '' : nested.prop}
</script>

<!--嵌套模板-->
<script type="text/html" id="template_item" data-nested="item">
    <li>
        <p>#{data.id}----#{data.name}</p>
        #{data.item === undefined ? '' : '<ul>' + nested.item + '</ul>'}
    </li>
</script>

<!--嵌套模板2-->
<script type="text/html" id="template_attr" data-nested="attr" data-alias="prop">
    <p>#{data.key}:#{data.value}</p>
</script>

填充模板：

$(function(){
    var data = {
        title: '标题',
        desc: '描述',
        item: [
            {
                id: 1,
                name: '我是子级'
            },
            {
                id: 2,
                name: '我是子级',
                item: [
                    {
                        id: 21,
                        name: '我是子级的子级'
                    },
                    {
                        id: 22,
                        name: '我是子级的子级'
                    }
                ]
            },
            {
                id: 3,
                name: '我是子级'
            }
        ],
        attr: [
            {
                key: 'name',
                value: '乔治'
            },
            {
                key: 'sex',
                value: '男'
            }
        ]
    };
    $('#target').html($('#template').template(data, {nested: '#template_item,#template_attr'}));
});

填充效果：

参数定义：

$('#template').template(data, {
    //嵌套内容模板，支持传入jQuery对象(单个或数组)/jQuery选择器字符串(单个或数组)
    nested: ['#template_item', '#template_attr', $('#template1'), $('#template2')]
    //是否清除未填充节点
    clean: true,
    //正则查询规则
    rule: /#{(.(?!#{))*[^\n\r\f\t\v\0]}/g
});
